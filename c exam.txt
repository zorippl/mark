#include<stdio.h>
int main(void)
{
	int a[50],n,i,pass=0,unpass=0;
	float avpass=0,avunpass=0,sumpass=0,sumunpass=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	for(i=0;i<n;i++)
	{
		if(a[i]>=60)
		{
			pass++;
			sumpass=sumpass+a[i];
		}
		else
		{
			unpass++;
			sumunpass=sumunpass+a[i];
		}
	}
	avpass=sumpass/pass;
	avunpass=sumunpass/unpass;
	printf("Pass:%d,%.1f  Fail:%d,%.1f",pass,avpass,unpass,avunpass);
}
